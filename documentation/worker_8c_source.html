743 static int read_internal(wss_server_t *server, wss_session_t *session, char *buffer) {
744  int n;
745 
746 #ifdef USE_OPENSSL
747  if ( NULL != server->ssl_ctx ) {
748  unsigned long err;
749 
750  n = SSL_read(session->ssl, buffer, server->config->size_buffer);
751  err = SSL_get_error(session->ssl, n);
752 
753  // There's no more to read from the kernel
754  if ( unlikely(err == SSL_ERROR_WANT_READ) ) {
755  n = 0;
756  } else
757 
758  // There's no space to write to the kernel, wait for filedescriptor.
759  if ( unlikely(err == SSL_ERROR_WANT_WRITE) ) {
760  WSS_log_debug("SSL_ERROR_WANT_WRITE");
761 
762  n = -2;
763  } else
764 
765  // Some error has occured.
766  if ( unlikely(err != SSL_ERROR_NONE && err != SSL_ERROR_ZERO_RETURN) ) {
767  char msg[1024];
768  while ( (err = ERR_get_error()) != 0 ) {
769  memset(msg, '\0', 1024);
770  ERR_error_string_n(err, msg, 1024);
771  WSS_log_error("SSL read failed: %s", msg);
772  }
773 
774  n = -1;
775  } else
776 
777  // If this was one of the error codes that we do accept, return 0
778  if ( unlikely(n < 0) ) {
779  n = 0;
780  }
781  } else {
782 #endif
783  do {
784  n = read(session->fd, buffer, server->config->size_buffer);
785  if (n == -1) {
786  if ( unlikely(errno == EINTR) ) {
787  errno = 0;
788  continue;
789  } else if ( unlikely(errno != EAGAIN && errno != EWOULDBLOCK) ) {
790  WSS_log_error("Read failed: %s", strerror(errno));
791  session->closing = true;
792  } else {
793  n = 0;
794  }
795  }
796  } while ( unlikely(0) );
797 #ifdef USE_OPENSSL
798  }
799 #endif
800 
801  return n;
802 }

811 static void handshake(wss_server_t *server, wss_session_t *session) {
812  int n;
813  wss_header_t *header;
814  wss_message_t *message;
815  enum HttpStatus_Code code;
816  char *buffer;
817 
818  if ( unlikely(NULL == (buffer = WSS_malloc(server->config->size_buffer)))) {
819  WSS_log_fatal("Unable to allocate buffer");
820  return;
821  }
822 
823  WSS_log_trace("Preparing client header");
824 
825  if ( unlikely(NULL != session->header) ) {
826  header = session->header;
827  session->header = NULL;
828  } else {
829  if ( unlikely(NULL == (header = WSS_malloc(sizeof(wss_header_t)))) ) {
830  WSS_log_fatal("Unable to allocate header");
831  return;
832  }
833 
834  header->content = NULL;
835  header->method = NULL;
836  header->version = NULL;
837  header->path = NULL;
838  header->host = NULL;
839  header->payload = NULL;
840  header->length = 0;
841  header->ws_version = 0;
842  header->ws_type = UNKNOWN;
843  header->ws_protocol = NULL;
844  header->ws_upgrade = NULL;
845  header->ws_connection = NULL;
846  header->ws_extensions = NULL;
847  header->ws_extensions_count = 0;
848  header->ws_origin = NULL;
849  header->ws_key = NULL;
850  header->ws_key1 = NULL;
851  header->ws_key2 = NULL;
852  header->ws_key3 = NULL;
853  }
854 
855  WSS_log_trace("Reading headers");
856 
857  // Continue reading until we get no bytes back
858  do {
859  n = read_internal(server, session, buffer);
860 
861  switch (n) {
862  // Wait for IO for either read or write on the filedescriptor
863  case -2:
864  session->header = header;
865 
866  session->event = WRITE;
867 
868  return;
869  // An error occured, notify client by writing back to it
870  case -1:
871  WSS_log_trace("Rejecting HTTP request due to being unable to read from client");
872 
873  message = http_response(header, HttpStatus_InternalServerError, "Unable to read from client");
874  WSS_free_header(header);
875 
876  if ( likely(NULL != message && WSS_SUCCESS == write_internal(session, message)) ) {
877  session->state = WRITING;
878  session->event = READ;
879  WSS_write(server, session);
880  }
881 
882  return;
883  case 0:
884  break;
885  default:
886  // Reallocate space for the header and copy buffer into it
887  if ( unlikely(NULL == (header->content = WSS_realloc((void **) &header->content, header->length*sizeof(char), (header->length+n+1)*sizeof(char)))) ) {
888  WSS_log_fatal("Unable to realloc header content");
889  return;
890  }
891  memcpy(header->content+header->length, buffer, n);
892  header->length += n;
893  memset(buffer, '\0', server->config->size_buffer);
894 
895  // Check if payload from client is too large for the server to handle.
896  // If so write error back to the client
897  if ( unlikely(header->length > (server->config->size_header+server->config->size_uri+server->config->size_payload)) ) {
898  WSS_log_trace("Rejecting HTTP request as client payload is too large for the server to handle");
899 
900  message = http_response(header, HttpStatus_PayloadTooLarge,
901  "The given payload is too large for the server to handle");
902  WSS_free_header(header);
903 
904  if ( likely(NULL != message && WSS_SUCCESS == write_internal(session, message)) ) {
905  session->state = WRITING;
906  session->event = READ;
907  WSS_write(server, session);
908  }
909 
910  return;
911  }
912  break;
913  }
914  } while ( likely(n != 0) );
915 
916  WSS_free((void **)&buffer);
917 
918  WSS_log_debug("Client header: \n%s", header->content);
919 
920  WSS_log_trace("Starting parsing header received from client");
921 
922  // Parsing HTTP header
923  // If header could not be parsed correctly, notify client with response.
924  if ( (code = WSS_parse_header(session->fd, header, server->config)) != HttpStatus_OK ) {
925  WSS_log_trace("Rejecting HTTP request due to header not being correct");
926 
927  message = http_response(header, code, (char *)HttpStatus_reasonPhrase(code));
928  WSS_free_header(header);
929 
930  if ( likely(NULL != message && WSS_SUCCESS == write_internal(session, message)) ) {
931  session->state = WRITING;
932  session->event = READ;
933  WSS_write(server, session);
934  }
935 
936  return;
937  }
938 
939  // Serve favicon
940  if (strlen(header->path) == 12 && strncmp(header->path, "/favicon.ico", 12) == 0) {
941  message = favicon_response(header, code, server->config);
942  if (NULL != message) {
943  WSS_free_header(header);
944 
945  WSS_log_trace("Serving a favicon to the client");
946 
947  // Find and serve favicon.
948  if ( likely(WSS_SUCCESS == write_internal(session, message)) ) {
949  session->state = WRITING;
950  session->event = READ;
951  WSS_write(server, session);
952  }
953  } else {
954  WSS_log_trace("Rejecting HTTP request due to favicon not being available");
955 
956  // Else notify client that favicon could not be found
957  code = HttpStatus_NotFound;
958  message = http_response(header, code, (char *)HttpStatus_reasonPhrase(code));
959  WSS_free_header(header);
960 
961  if ( likely(NULL != message && WSS_SUCCESS == write_internal(session, message)) ) {
962  session->state = WRITING;
963  session->event = READ;
964  WSS_write(server, session);
965  }
966  }
967 
968  return;
969  }
970 
971  WSS_log_trace("Header successfully parsed");
972 
973  // Create Upgrade HTTP header based on clients header
974  code = WSS_upgrade_header(header, server->config, server->re);
975  switch (code) {
976  case HttpStatus_UpgradeRequired:
977  WSS_log_trace("Rejecting HTTP request as the service requires use of the Websocket protocol.");
978  message = upgrade_response(header, code,
979  "This service requires use of the Websocket protocol.");
980  break;
981  case HttpStatus_NotImplemented:
982  WSS_log_trace("Rejecting HTTP request as Websocket protocol is not yet implemented");
983  message = http_response(header, code,
984  "Websocket protocol is not yet implemented");
985  break;
986  case HttpStatus_SwitchingProtocols:
987  message = handshake_response(header, code);
988  break;
989  case HttpStatus_NotFound:
990  WSS_log_trace("Rejecting HTTP request as the page requested was not found.");
991  message = http_response(header, code,
992  (char *)HttpStatus_reasonPhrase(code));
993  break;
994  case HttpStatus_Forbidden:
995  WSS_log_trace("Rejecting HTTP request as the origin is not allowed to establish a websocket connection.");
996  message = http_response(header, code,
997  "The origin is not allowed to establish a websocket connection.");
998  break;
999  default:
1000  WSS_log_trace("Rejecting HTTP request as server was unable to parse http header as websocket request");
1001  message = http_response(header, code,
1002  (char *)HttpStatus_reasonPhrase(code));
1003  break;
1004  }
1005 
1006  // If code status isnt switching protocols, we notify client with a HTTP error
1007  if (code != HttpStatus_SwitchingProtocols) {
1008  WSS_log_trace("Unable to establish websocket connection");
1009 
1010  if ( likely(WSS_SUCCESS == write_internal(session, message)) ) {
1011  session->state = WRITING;
1012  session->event = READ;
1013  WSS_write(server, session);
1014  }
1015 
1016  WSS_free_header(header);
1017 
1018  return;
1019  }
1020 
1021  // Use echo protocol if none was chosen
1022  if (NULL == header->ws_protocol) {
1023  header->ws_protocol = WSS_find_subprotocol("echo");
1024  }
1025 
1026  // Notify websocket protocol of the connection
1027  header->ws_protocol->connect(session->fd, session->ip, session->port, header->path, header->cookies);
1028 
1029  // Set session as fully handshaked
1030  session->handshaked = true;
1031  session->header = header;
1032 
1033  if ( likely(WSS_SUCCESS == write_internal(session, message)) ) {
1034  session->state = WRITING;
1035  session->event = READ;
1036  WSS_write(server, session);
1037  }
1038 }

1047 void WSS_read(wss_server_t *server, wss_session_t *session) {
1048  int n;
1049  size_t len;
1050  uint16_t code;
1051  wss_frame_t *frame;
1052  char *msg;
1053  bool closing = false;
1054  size_t offset = 0, prev_offset = 0;
1055  wss_frame_t **frames = NULL;
1056  char *payload = NULL;
1057  size_t payload_length = 0;
1058  size_t frames_length = 0;
1059  char *message;
1060  size_t message_length;
1061  wss_message_t *m;
1062  size_t i, j, k;
1063  size_t msg_length = 0;
1064  size_t msg_offset = 0;
1065  size_t starting_frame = 0;
1066  bool fragmented = false;
1067  char *buffer;
1068 
1069  // If no initial header has been seen for the session, the websocket
1070  // handshake is yet to be made.
1071  if ( unlikely(! session->handshaked) ){
1072  WSS_log_trace("Doing websocket handshake");
1073  handshake(server, session);
1074  return;
1075  }
1076 
1077  WSS_log_trace("Starting initial steps to read from client");
1078 
1079  if ( unlikely(NULL == (buffer = WSS_malloc(server->config->size_buffer)))) {
1080  WSS_log_error("Unable to allocate buffer");
1081  session->closing = true;
1082  return;
1083  }
1084 
1085  // Use earlier payload
1086  payload = session->payload;
1087  payload_length = session->payload_length;
1088  offset = session->offset;
1089  frames = session->frames;
1090  frames_length = session->frames_length;
1091  session->payload = NULL;
1092  session->payload_length = 0;
1093  session->offset = 0;
1094  session->frames = NULL;
1095  session->frames_length = 0;
1096 
1097  // If handshake has been made, we can read the websocket frames from
1098  // the connection
1099  do {
1100  n = read_internal(server, session, buffer);
1101 
1102  switch (n) {
1103  // Wait for IO for either read or write on the filedescriptor
1104  case -2:
1105  WSS_log_trace("Detected that server needs further IO to complete the reading");
1106  session->payload = payload;
1107  session->payload_length = payload_length;
1108 
1109  session->event = WRITE;
1110 
1111  return;
1112  // An error occured
1113  case -1:
1114  WSS_free((void **) &payload);
1115 
1116  frame = WSS_closing_frame(CLOSE_UNEXPECTED, NULL);
1117  message_length = WSS_stringify_frame(frame, &message);
1118  WSS_free_frame(frame);
1119 
1120  if ( unlikely(NULL == (m = WSS_malloc(sizeof(wss_message_t)))) ) {
1121  WSS_log_error("Unable to allocate the message structure");
1122  WSS_free((void **) &message);
1123  session->closing = true;
1124  return;
1125  }
1126  m->msg = message;
1127  m->length = message_length;
1128  m->framed = true;
1129 
1130  if ( likely(WSS_SUCCESS == write_internal(session, m)) ) {
1131  session->state = WRITING;
1132  session->event = READ;
1133  WSS_write(server, session);
1134  }
1135  return;
1136  // No new data received
1137  case 0:
1138  break;
1139  // Empty buffer into payload
1140  default:
1141  if ( unlikely(NULL == (payload = WSS_realloc((void **) &payload, payload_length*sizeof(char), (payload_length+n+1)*sizeof(char)))) ) {
1142  WSS_log_error("Unable to reallocate the payload");
1143  session->closing = true;
1144  return;
1145  }
1146 
1147  memcpy(payload+payload_length, buffer, n);
1148  payload_length += n;
1149  memset(buffer, '\0', server->config->size_buffer);
1150 
1151  }
1152  } while ( likely(n != 0) );
1153 
1154  // Release memory used for buffer
1155  WSS_free((void **) &buffer);
1156 
1157  WSS_log_trace("Payload from client was read. Continues flow by parsing frames.");
1158 
1159  // Parse the payload into websocket frames
1160  do {
1161  prev_offset = offset;
1162 
1163  if ( unlikely(NULL == (frame = WSS_parse_frame(payload, payload_length, &offset))) ) {
1164  WSS_log_trace("Unable to parse frame");
1165  WSS_free((void **) &payload);
1166  session->closing = true;
1167  return;
1168  }
1169 
1170  // Check if we were forced to read beyond the payload to create a full frame
1171  if ( unlikely(offset > payload_length) ) {
1172  WSS_log_trace("Detected that data was missing in order to complete frame, will wait for more");
1173 
1174  WSS_free_frame(frame);
1175 
1176  session->payload = payload;
1177  session->payload_length = payload_length;
1178  session->offset = prev_offset;
1179  session->frames = frames;
1180  session->frames_length = frames_length;
1181  session->event = READ;
1182 
1183  return;
1184  }
1185 
1186  // If no extension is negotiated, the rsv bits must not be used
1187  if ( unlikely(NULL == session->header->ws_extensions && (frame->rsv1 || frame->rsv2 || frame->rsv3)) ) {
1188  WSS_log_trace("Protocol Error: rsv bits must not be set without using extensions");
1189  WSS_free_frame(frame);
1190  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1191  } else
1192 
1193  // If opcode is unknown
1194  if ( unlikely((frame->opcode >= 0x3 && frame->opcode <= 0x7) ||
1195  (frame->opcode >= 0xB && frame->opcode <= 0xF)) ) {
1196  WSS_log_trace("Type Error: Unknown upcode");
1197  WSS_free_frame(frame);
1198  frame = WSS_closing_frame(CLOSE_TYPE, NULL);
1199  } else
1200 
1201  // Server expects all received data to be masked
1202  if ( unlikely(! frame->mask) ) {
1203  WSS_log_trace("Protocol Error: Client message should always be masked");
1204  WSS_free_frame(frame);
1205  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1206  } else
1207 
1208  // Control frames cannot be fragmented
1209  if ( unlikely(! frame->fin && frame->opcode >= 0x8 && frame->opcode <= 0xA) ) {
1210  WSS_log_trace("Protocol Error: Control frames cannot be fragmented");
1211  WSS_free_frame(frame);
1212  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1213  } else
1214 
1215  // Check that frame is not too large
1216  if ( unlikely(frame->payloadLength > server->config->size_frame) ) {
1217  WSS_log_trace("Protocol Error: Control frames cannot be fragmented");
1218  WSS_free_frame(frame);
1219  frame = WSS_closing_frame(CLOSE_BIG, NULL);
1220  } else
1221 
1222  // Control frames cannot have a payload length larger than 125 bytes
1223  if ( unlikely(frame->opcode >= 0x8 && frame->opcode <= 0xA && frame->payloadLength > 125) ) {
1224  WSS_log_trace("Protocol Error: Control frames cannot have payload larger than 125 bytes");
1225  WSS_free_frame(frame);
1226  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1227  } else
1228 
1229  // In HYBI10 specification the most significant bit must not be set
1230  if ( unlikely((session->header->ws_type == HYBI10 || session->header->ws_type == HYBI07) && frame->payloadLength & ((uint64_t)1 << (sizeof(uint64_t)*8-1))) ) {
1231  WSS_log_trace("Protocol Error: Frame payload length must not use MSB");
1232  WSS_free_frame(frame);
1233  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1234  } else
1235 
1236  // Close frame
1237  if ( unlikely(frame->opcode == CLOSE_FRAME) ) {
1238  // A code of 2 byte must be present if there is any application data for closing frame
1239  if ( unlikely(frame->applicationDataLength > 0 && frame->applicationDataLength < 2) ) {
1240  WSS_log_trace("Protocol Error: Closing frame with payload too small bytewise error code");
1241  WSS_free_frame(frame);
1242  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1243  } else
1244 
1245 
1246  // The payload after the code, must be valid UTF8
1247  if ( unlikely(frame->applicationDataLength >= 2 && ! utf8_check(frame->payload+2, frame->applicationDataLength-2)) ) {
1248  WSS_log_trace("Protocol Error: Payload of error frame must be valid UTF8.");
1249  WSS_free_frame(frame);
1250  frame = WSS_closing_frame(CLOSE_UTF8, NULL);
1251  } else
1252 
1253  // Check status code is within valid range
1254  if (frame->applicationDataLength >= 2) {
1255  // Copy code
1256  memcpy(&code, frame->payload, sizeof(uint16_t));
1257  code = ntohs(code);
1258 
1259  WSS_log_debug("Closing frame code: %d", code);
1260 
1261  // Current rfc6455 codes
1262  if ( unlikely(code < 1000 || (code >= 1004 && code <= 1006) || (code >= 1015 && code < 3000) || code >= 5000) ) {
1263  WSS_log_trace("Protocol Error: Closing frame has invalid error code");
1264  WSS_free_frame(frame);
1265  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1266  }
1267  }
1268  } else
1269 
1270  // Pong
1271  if ( unlikely(frame->opcode == PONG_FRAME) ) {
1272  WSS_log_trace("Pong received");
1273 
1274  WSS_free_frame(frame);
1275 
1276  continue;
1277  } else
1278 
1279  // Ping
1280  if ( unlikely(frame->opcode == PING_FRAME) ) {
1281  WSS_log_trace("Ping received");
1282  frame = WSS_pong_frame(frame);
1283  }
1284 
1285  if ( unlikely(NULL == (frames = WSS_realloc((void **) &frames, frames_length*sizeof(wss_frame_t *),
1286  (frames_length+1)*sizeof(wss_frame_t *)))) ) {
1287  WSS_log_error("Unable to reallocate frames");
1288  session->closing = true;
1289  return;
1290  }
1291  frames[frames_length] = frame;
1292  frames_length += 1;
1293 
1294  // Check if frame count is exceeded
1295  if ( unlikely(frames_length > server->config->max_frames) ) {
1296  WSS_free_frame(frame);
1297  frame = WSS_closing_frame(CLOSE_BIG, NULL);
1298  frames[frames_length-1] = frame;
1299  }
1300 
1301  // Close
1302  if ( unlikely(frame->opcode == CLOSE_FRAME) ) {
1303  closing = true;
1304  WSS_log_trace("Stopping frame validation as closing frame was parsed");
1305  break;
1306  }
1307  } while ( likely(offset < payload_length) );
1308 
1309  WSS_free((void **) &payload);
1310 
1311  WSS_log_trace("A total of %lu frames was parsed.", frames_length);
1312 
1313  WSS_log_trace("Starting frame validation");
1314 
1315  // Validating frames.
1316  for (i = 0; likely(i < frames_length); i++) {
1317  // If we are not processing a fragmented set of frames, expect the
1318  // opcode different from the continuation frame.
1319  if ( unlikely(! fragmented && (frames[i]->opcode == CONTINUATION_FRAME)) ) {
1320  WSS_log_trace("Protocol Error: continuation opcode used in non-fragmented message");
1321  for (j = i; likely(j < frames_length); j++) {
1322  WSS_free_frame(frames[j]);
1323  }
1324  frames[i] = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1325  frames_length = i+1;
1326  closing = true;
1327  break;
1328  }
1329 
1330  // If we receive control frame within fragmented frames.
1331  if ( unlikely(fragmented && (frames[i]->opcode >= 0x8 && frames[i]->opcode <= 0xA)) ) {
1332  WSS_log_trace("Received control frame within fragmented message");
1333 
1334  frame = frames[i];
1335  // If we received a closing frame substitue the fragment with the
1336  // closing frame and end validation
1337  if (frames[i]->opcode == CLOSE_FRAME) {
1338  WSS_log_trace("Stopping further validation of fragmented message as a closing frame was detected");
1339 
1340  for (j = starting_frame; likely(j < frames_length); j++) {
