116 threadpool_t *threadpool_create(int thread_count, int queue_size,
117  int thread_size, int flags) {
118  threadpool_t *pool;
119  int i;
120 
121  /* TODO: Check for negative or otherwise very big input parameters */
122 
123  if ((pool = (threadpool_t *)WSS_malloc(sizeof(threadpool_t))) == NULL) {
124  goto err;
125  }
126 
127  /* Initialize */
128  pool->thread_count = 0;
129  pool->queue_size = queue_size;
130  pool->head = pool->tail = pool->count = 0;
131  pool->shutdown = pool->started = 0;
132 
133  /* Allocate attributes */
134  pthread_attr_init(&pool->attr);
135  pthread_attr_setstacksize(&pool->attr, thread_size);
136 
137  /* Allocate thread and task queue */
138  pool->threads = (pthread_t *) WSS_malloc(sizeof(pthread_t) * thread_count);
139  pool->queue = (threadpool_task_t *) WSS_malloc(sizeof(threadpool_task_t) * queue_size);
140 
141  /* Initialize mutex and conditional variable first */
142  if ( (pthread_mutex_init(&(pool->lock), NULL) != 0) ||
143  (pthread_cond_init(&(pool->notify), NULL) != 0) ||
144  (pool->threads == NULL) ||
145  (pool->queue == NULL)) {
146  goto err;
147  }
148 
149  /* Start worker threads */
150  for (i = 0; i < thread_count; i++) {
151  if (pthread_create(&(pool->threads[i]), &pool->attr, threadpool_thread,
152  (void*) pool) != 0) {
153  threadpool_destroy(pool, 0);
154  return NULL;
155  }
156  pool->thread_count++;
157  pool->started++;
158  }
159 
160  return pool;
161 
162 err:
163  if (pool) {
164  threadpool_free(pool);
165  }
166  return NULL;
167 }

169 int threadpool_add(threadpool_t *pool, void (*function)(void *),
170  void *argument, int flags){
171  int err = 0;
172  int next;
173 
174  (void) flags;
175 
176  if (NULL == pool || NULL == function) {
177  return threadpool_invalid;
178  }
179 
180  if (pthread_mutex_lock(&(pool->lock)) != 0) {
181  return threadpool_lock_failure;
182  }
183 
184  next = pool->tail + 1;
185  next = (next == pool->queue_size) ? 0 : next;
186 
187  do {
188  /* Are we full ? */
189  if (pool->count == pool->queue_size) {
190  err = threadpool_queue_full;
191  break;
192  }
193 
194  /* Are we shutting down ? */
195  if (pool->shutdown) {
196  err = threadpool_shutdown;
197  break;
198  }
199 
200  /* Add task to queue */
201  pool->queue[pool->tail].function = function;
202  pool->queue[pool->tail].argument = argument;
203  pool->tail = next;
204  pool->count += 1;
205 
206  /* pthread_cond_broadcast */
207  if (pthread_cond_signal(&(pool->notify)) != 0) {
208  err = threadpool_lock_failure;
209  break;
210  }
211  } while(0);
212 
213  if (pthread_mutex_unlock(&pool->lock) != 0) {
214  err = threadpool_lock_failure;
215  }
216 
217  return err;
218 }

220 int threadpool_destroy(threadpool_t *pool, int flags) {
221  int i, err = 0;
222 
223  if (NULL == pool) {
224  return threadpool_invalid;
225  }
226 
227  if (pthread_mutex_lock(&(pool->lock)) != 0) {
228  return threadpool_lock_failure;
229  }
230 
231  do {
232  /* Already shutting down */
233  if (pool->shutdown) {
234  err = threadpool_shutdown;
235  break;
236  }
237 
238  pool->shutdown = (flags & threadpool_graceful) ?
239  graceful_shutdown : immediate_shutdown;
240 
241  /* Wake up all worker threads */
242  if ( (pthread_cond_broadcast(&(pool->notify)) != 0) ||
243  (pthread_mutex_unlock(&(pool->lock)) != 0) ) {
244  err = threadpool_lock_failure;
245  break;
246  }
247 
248  /* Join all worker thread */
249  for (i = 0; i < pool->thread_count; i++) {
250  if (pthread_join(pool->threads[i], NULL) != 0) {
251  err = threadpool_thread_failure;
252  }
253  }
254  } while(0);
255 
256  /* Only if everything went well do we deallocate the pool */
257  if (!err) {
258  threadpool_free(pool);
259  }
260  return err;
261 }

263 int threadpool_free(threadpool_t *pool) {
264  if(NULL == pool || pool->started > 0) {
265  return -1;
266  }
267 
268  /* Did we manage to allocate ? */
269  if (pool->threads) {
270  pthread_attr_destroy(&pool->attr);
271  WSS_free((void **) &pool->threads);
272  WSS_free((void **) &pool->queue);
273 
274  /* Because we allocate pool->threads after initializing the
275  mutex and condition variable, we're sure they're
276  initialized. Let's lock the mutex just in case. */
277  pthread_mutex_lock(&(pool->lock));
278  pthread_mutex_destroy(&(pool->lock));
279  pthread_cond_destroy(&(pool->notify));
280  }
281 
282  WSS_free((void **) &pool);
283  return 0;
284 }

287 static void *threadpool_thread(void *arguments) {
288  threadpool_t *pool = (threadpool_t *)arguments;
289  threadpool_task_t task;
290 
291 #ifdef USE_RPMALLOC
292  rpmalloc_thread_initialize();
293 #endif
294 
295  for (;;) {
296  /* Lock must be taken to wait on conditional variable */
297  pthread_mutex_lock(&(pool->lock));
298 
299  /* Wait on condition variable, check for spurious wakeups.
300  When returning from pthread_cond_wait(), we own the lock. */
301  while ( (pool->count == 0) && (!pool->shutdown) ) {
302  pthread_cond_wait(&(pool->notify), &(pool->lock));
303  }
304 
305  if ( (pool->shutdown == immediate_shutdown) ||
306  ((pool->shutdown == graceful_shutdown) && (pool->count == 0)) ) {
307  break;
308  }
309 
310  /* Grab our task */
311  task.function = pool->queue[pool->head].function;
312  task.argument = pool->queue[pool->head].argument;
313  pool->head += 1;
314  pool->head = (pool->head == pool->queue_size) ? 0 : pool->head;
315  pool->count -= 1;
316 
317  /* Unlock */
318  pthread_mutex_unlock(&(pool->lock));
319  /* Get to work */
320  (*(task.function))(task.argument);
321  }
322 
323  pool->started--;
324 
325  pthread_mutex_unlock(&(pool->lock));
326 
327 #ifdef USE_RPMALLOC
328  rpmalloc_thread_finalize();
329 #endif
330 
331  pthread_exit(NULL);
332  return NULL;
333 }
